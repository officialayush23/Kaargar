-- ======== Minimal MVP schema for KAARGAR ========
-- Assumes PostGIS installed for geography types.

-- USERS (core identity: id MUST match Supabase auth sub)
CREATE TABLE IF NOT EXISTS public.users (
  id uuid PRIMARY KEY,                             -- must equal auth.sub
  email text UNIQUE NOT NULL,
  name text,
  phone text,
  phone_masked text,
  role text[] NOT NULL DEFAULT ARRAY['customer']::text[],
  created_at timestamptz DEFAULT now(),
  location geography(Point,4326),                  -- worker/customer location
  is_active boolean DEFAULT true,
  is_flagged boolean DEFAULT false,
  rating_avg numeric(3,2) DEFAULT 0,
  rating_count integer DEFAULT 0,
  cancellation_count integer DEFAULT 0,
  complaints_count integer DEFAULT 0,
  trouble_score numeric DEFAULT 0,
  tags text[] DEFAULT '{}'::text[],
  metadata jsonb DEFAULT '{}'::jsonb,
  profile_completed boolean DEFAULT false
);

CREATE INDEX IF NOT EXISTS idx_users_location ON public.users USING GIST (location);
CREATE INDEX IF NOT EXISTS idx_users_tags ON public.users USING GIN (tags);

-- WORKER PROFILES (one-to-one)
CREATE TABLE IF NOT EXISTS public.worker_profiles (
  user_id uuid PRIMARY KEY REFERENCES public.users(id) ON DELETE CASCADE,
  professions text[] DEFAULT '{}'::text[],
  skills text[] DEFAULT '{}'::text[],
  hourly_rate_cents bigint,            -- store money in cents
  min_hourly_rate_cents bigint,
  max_hourly_rate_cents bigint,
  accepts_remote boolean DEFAULT false,
  bio text,
  availability jsonb DEFAULT '{}'::jsonb,
  documents jsonb DEFAULT '{}'::jsonb, -- {type: storage_key}
  search_radius_km integer DEFAULT 10,
  max_concurrent_jobs integer DEFAULT 1,
  kyc_status text DEFAULT 'none',
  created_at timestamptz DEFAULT now()
);
CREATE INDEX IF NOT EXISTS idx_worker_professions_gin ON public.worker_profiles USING GIN (professions);
CREATE INDEX IF NOT EXISTS idx_worker_skills_gin ON public.worker_profiles USING GIN (skills);

-- JOBS
CREATE TABLE IF NOT EXISTS public.jobs (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  customer_id uuid NOT NULL REFERENCES public.users(id) ON DELETE CASCADE,
  title text NOT NULL,
  description text,
  category text,
  tags text[] DEFAULT '{}'::text[],
  location geography(Point,4326),
  is_remote boolean DEFAULT false,
  price_type text NOT NULL CHECK (price_type IN ('fixed','hourly','bidding')),
  price_min_cents bigint,
  price_max_cents bigint,
  scheduled_start timestamptz,
  scheduled_end timestamptz,
  status text NOT NULL DEFAULT 'draft',   -- draft, posted, assigned, in_progress, completed, cancelled
  assigned_worker_id uuid REFERENCES public.users(id),
  created_at timestamptz DEFAULT now(),
  expires_at timestamptz,
  metadata jsonb DEFAULT '{}'::jsonb
);
CREATE INDEX IF NOT EXISTS idx_jobs_location ON public.jobs USING GIST (location);
CREATE INDEX IF NOT EXISTS idx_jobs_category ON public.jobs (category);

-- BIDS
CREATE TABLE IF NOT EXISTS public.bids (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id uuid NOT NULL REFERENCES public.jobs(id) ON DELETE CASCADE,
  worker_id uuid NOT NULL REFERENCES public.users(id) ON DELETE CASCADE,
  amount_cents bigint,
  message text,
  status text NOT NULL DEFAULT 'pending',
  created_at timestamptz DEFAULT now()
);
CREATE INDEX IF NOT EXISTS idx_bids_job ON public.bids(job_id);

-- ASSIGNMENTS (one assignment per job)
CREATE TABLE IF NOT EXISTS public.assignments (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id uuid NOT NULL UNIQUE REFERENCES public.jobs(id) ON DELETE CASCADE,
  worker_id uuid NOT NULL REFERENCES public.users(id),
  scheduled_start timestamptz NOT NULL,
  scheduled_end timestamptz NOT NULL,
  status text NOT NULL DEFAULT 'pending_worker_accept',
  created_at timestamptz DEFAULT now(),
  accepted_at timestamptz,
  cancelled_at timestamptz
);
CREATE INDEX IF NOT EXISTS idx_assignments_worker_time ON public.assignments(worker_id, scheduled_start, scheduled_end);

-- PAYMENTS (use cents)
CREATE TABLE IF NOT EXISTS public.payments (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id uuid REFERENCES public.jobs(id),
  payer_id uuid REFERENCES public.users(id),
  payee_id uuid REFERENCES public.users(id),
  provider text,
  provider_payment_id text,
  amount_total_cents bigint NOT NULL DEFAULT 0,      -- integer cents
  platform_fee_cents bigint DEFAULT 0,
  amount_to_worker_cents bigint DEFAULT 0,
  status text NOT NULL DEFAULT 'initiated',         -- initiated, captured, released, refunded
  platform_fee_pct numeric(5,2),
  payout_status text DEFAULT 'not_initiated',
  payout_provider_id text,
  created_at timestamptz DEFAULT now(),
  metadata jsonb DEFAULT '{}'::jsonb
);
CREATE INDEX IF NOT EXISTS idx_payments_job ON public.payments(job_id);

-- ESCROW table (optional but clear)
CREATE TABLE IF NOT EXISTS public.escrows (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  payment_id uuid REFERENCES public.payments(id) ON DELETE CASCADE,
  job_id uuid REFERENCES public.jobs(id),
  amount_cents bigint NOT NULL,
  status text DEFAULT 'held',  -- held|released|refunded
  created_at timestamptz DEFAULT now(),
  released_at timestamptz
);

-- WALLETS & TRANSACTIONS
CREATE TABLE IF NOT EXISTS public.wallets (
  user_id uuid PRIMARY KEY REFERENCES public.users(id) ON DELETE CASCADE,
  balance_cents bigint DEFAULT 0,
  reserved_cents bigint DEFAULT 0,
  updated_at timestamptz DEFAULT now()
);
CREATE TABLE IF NOT EXISTS public.wallet_transactions (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id uuid REFERENCES public.users(id),
  amount_cents bigint NOT NULL,
  kind text NOT NULL,                 -- reserved, payout, platform_fee, platform_hold, refund
  reference_id uuid,
  metadata jsonb DEFAULT '{}'::jsonb,
  created_at timestamptz DEFAULT now()
);
CREATE INDEX IF NOT EXISTS idx_wallet_tx_user ON public.wallet_transactions(user_id);

-- PAYOUTS
CREATE TABLE IF NOT EXISTS public.payouts (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  payment_id uuid REFERENCES public.payments(id),
  worker_id uuid REFERENCES public.users(id),
  amount_cents bigint NOT NULL,
  platform_fee_cents bigint,
  provider text,
  provider_payout_id text,
  status text DEFAULT 'created',  -- created|success|failed
  attempted_at timestamptz,
  created_at timestamptz DEFAULT now(),
  error text
);

-- KYC documents
CREATE TABLE IF NOT EXISTS public.kyc_documents (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id uuid NOT NULL REFERENCES public.users(id),
  doc_type text NOT NULL,
  storage_key text NOT NULL,    -- Supabase storage path
  content_hash text,            -- optional
  status text NOT NULL DEFAULT 'uploaded', -- uploaded|approved|rejected
  uploaded_at timestamptz DEFAULT now(),
  reviewed_by uuid REFERENCES public.users(id),
  reviewed_at timestamptz,
  reviewed_notes text
);

-- RATINGS
CREATE TABLE IF NOT EXISTS public.ratings (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id uuid REFERENCES public.jobs(id),
  from_user_id uuid NOT NULL REFERENCES public.users(id),
  to_user_id uuid NOT NULL REFERENCES public.users(id),
  rating smallint CHECK (rating BETWEEN 1 AND 5),
  review_text text,
  created_at timestamptz DEFAULT now()
);
CREATE INDEX IF NOT EXISTS idx_ratings_to ON public.ratings(to_user_id);

-- DISPUTES
CREATE TABLE IF NOT EXISTS public.disputes (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id uuid REFERENCES public.jobs(id),
  opened_by uuid REFERENCES public.users(id),
  reason text,
  status text DEFAULT 'open',   -- open|resolved
  resolution text,
  created_at timestamptz DEFAULT now(),
  resolved_at timestamptz
);

-- MEDIA
CREATE TABLE IF NOT EXISTS public.media_attachments (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  owner_id uuid REFERENCES public.users(id),
  job_id uuid REFERENCES public.jobs(id),
  storage_key text NOT NULL,
  filename text,
  mime_type text,
  size_bytes integer,
  tag text,
  created_at timestamptz DEFAULT now()
);

-- CHATS / MESSAGES
CREATE TABLE IF NOT EXISTS public.chats (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  job_id uuid REFERENCES public.jobs(id),
  created_at timestamptz DEFAULT now()
);
CREATE TABLE IF NOT EXISTS public.messages (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  chat_id uuid REFERENCES public.chats(id) ON DELETE CASCADE,
  sender_id uuid REFERENCES public.users(id),
  body text,
  attachments jsonb DEFAULT '[]'::jsonb,
  read_by jsonb DEFAULT '[]'::jsonb,
  created_at timestamptz DEFAULT now()
);

-- EVENTS & WEBHOOKS (for idempotency)
CREATE TABLE IF NOT EXISTS public.events (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  entity_type text,
  entity_id uuid,
  event_type text,
  payload jsonb DEFAULT '{}'::jsonb,
  created_at timestamptz DEFAULT now()
);
CREATE TABLE IF NOT EXISTS public.webhook_logs (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  provider text,
  provider_event_id text, -- store unique provider event id for dedupe
  provider_event jsonb,
  received_at timestamptz DEFAULT now(),
  UNIQUE(provider, provider_event_id)
);

-- FEE_RULES (platform fees)
CREATE TABLE IF NOT EXISTS public.fee_rules (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  scope text NOT NULL CHECK (scope IN ('global','company','category','worker')),
  scope_id uuid,
  category text,
  pct numeric(5,2) NOT NULL,
  active boolean DEFAULT true,
  precedence integer DEFAULT 100,
  created_at timestamptz DEFAULT now()
);

-- PUSH TOKENS / NOTIFS (minimal)
CREATE TABLE IF NOT EXISTS public.push_tokens (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id uuid REFERENCES public.users(id),
  provider text NOT NULL,
  token text NOT NULL,
  platform text,
  meta jsonb DEFAULT '{}'::jsonb,
  created_at timestamptz DEFAULT now()
);
CREATE TABLE IF NOT EXISTS public.push_notifications (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  recipient_id uuid REFERENCES public.users(id),
  sender_id uuid REFERENCES public.users(id),
  channel text NOT NULL CHECK (channel IN ('push','email','sms','inapp')),
  title text,
  body text,
  data jsonb DEFAULT '{}'::jsonb,
  status text NOT NULL DEFAULT 'queued',
  created_at timestamptz DEFAULT now()
);
